```python
def solution(heights):
    answer = []
    
    for i in range(len(heights)-1, -1, -1):
        flag = False
        for j in range(i-1, -1, -1):
            print(j)
            if heights[i] < heights[j]:
                flag = True
                answer = [j + 1] + answer
                print(answer)
                break
        if not flag:
            answer = [0] + answer

    return answer
```
